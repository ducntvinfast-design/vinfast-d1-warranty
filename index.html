function doGet(e){ const action = e.parameter.action || 'getAll'; try { if(action === 'getAll') return jsonResp(getAllData()); if(action === 'ping') return jsonResp({ok:true, time: new Date().toISOString()}); } catch(err) { return jsonResp({error: err.message}); } } function doPost(e){ try { const body = JSON.parse(e.postData.contents); const type = body.type; const data = body.data; if(type === 'ycbh') return jsonResp(upsertRow(SHEET_YCBH, HEADERS_YCBH, data, 'id')); if(type === 'kho') return jsonResp(upsertRow(SHEET_KHO, HEADERS_KHO, data, 'id')); if(type === 'kt') return jsonResp(upsertRow(SHEET_KT, HEADERS_KT, data, 'ycbhId')); if(type === 'deleteYCBH') return jsonResp(deleteRow(SHEET_YCBH, 'id', data.id)); if(type === 'bulkYCBH'){ data.forEach(row => upsertRow(SHEET_YCBH, HEADERS_YCBH, row, 'id')); return jsonResp({success:true, count: data.length}); } if(type === 'bulkKho'){ data.forEach(row => upsertRow(SHEET_KHO, HEADERS_KHO, row, 'id')); return jsonResp({success:true, count: data.length}); } return jsonResp({error:'Unknown type: '+type}); } catch(err) { return jsonResp({error: err.message}); } } function getAllData(){ return { ycbh: readSheet(SHEET_YCBH, HEADERS_YCBH), kho: readSheet(SHEET_KHO, HEADERS_KHO), kt: readSheetAsMap(SHEET_KT, HEADERS_KT, 'ycbhId'), }; } function initSheets(){ const ss = SpreadsheetApp.openById(SHEET_ID); ensureSheet(ss, SHEET_YCBH, HEADERS_YCBH, '#1565C0'); ensureSheet(ss, SHEET_KHO, HEADERS_KHO, '#2E7D32'); ensureSheet(ss, SHEET_KT, HEADERS_KT, '#6A1B9A'); return 'Sheets initialized successfully!'; } function ensureSheet(ss, name, headers, color){ let ws = ss.getSheetByName(name); if(!ws){ ws = ss.insertSheet(name); if(color) ws.setTabColor(color); } // Write headers if row 1 is empty if(ws.getRange(1,1).getValue() === ''){ ws.getRange(1, 1, 1, headers.length).setValues([headers]); ws.getRange(1, 1, 1, headers.length) .setBackground('#37474F').setFontColor('#FFFFFF') .setFontWeight('bold').setFontSize(11); ws.setFrozenRows(1); ws.autoResizeColumns(1, headers.length); } return ws; } function readSheet(sheetName, headers){ const ss = SpreadsheetApp.openById(SHEET_ID); const ws = ss.getSheetByName(sheetName); if(!ws) return []; const last = ws.getLastRow(); if(last < 2) return []; const data = ws.getRange(2, 1, last-1, headers.length).getValues(); return data .filter(row => row[0] !== '' && row[0] !== null) .map(row => { const obj = {}; headers.forEach((h,i) => { let val = row[i]; if(val instanceof Date) val = Utilities.formatDate(val, 'Asia/Ho_Chi_Minh', 'yyyy-MM-dd'); obj[h] = val === null || val === undefined ? '' : val; }); return obj; }); } function readSheetAsMap(sheetName, headers, keyField){ const rows = readSheet(sheetName, headers); const map = {}; rows.forEach(r => { if(r[keyField]) map[r[keyField]] = r; }); return map; } function upsertRow(sheetName, headers, data, keyField){ const ss = SpreadsheetApp.openById(SHEET_ID); let ws = ss.getSheetByName(sheetName); if(!ws) ws = ensureSheet(ss, sheetName, headers, null); data.updatedAt = new Date().toISOString(); const rowData = headers.map(h => data[h] !== undefined ? data[h] : ''); // Find existing row const keyVal = data[keyField]; if(keyVal){ const last = ws.getLastRow(); if(last >= 2){ const keyCol = headers.indexOf(keyField) + 1; const keyRange = ws.getRange(2, keyCol, last-1, 1).getValues(); for(let i=0; i=0; i--){ if(String(keyRange[i][0]) === String(keyVal)){ ws.deleteRow(i+2); return {success:true, deleted:true}; } } return {success:false, error:'Row not found'}; } function jsonResp(data){ return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
